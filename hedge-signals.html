"""
MacroFX Hedge Signal Poller
Poll /api/hedge-signals every 30 min and act on ACTIVE signals.
"""
import requests, time, logging

MACROFX_URL = "https://your-railway-url.up.railway.app"
POLL_INTERVAL = 30 * 60  # seconds

def fetch_signals():
    r = requests.get(f"{MACROFX_URL}/api/hedge-signals", timeout=10)
    r.raise_for_status()
    return r.json()

def handle_signal(sig):
    pair_a = sig["pair_a"]
    pair_b = sig["pair_b"]
    dir_a  = sig["direction_a"]  # "LONG" or "SHORT"
    dir_b  = sig["direction_b"]
    z      = sig["z_score"]
    score  = sig["hedge_score"]
    regime = sig["regime"]

    logging.info(
        f"Signal: {pair_a} {dir_a} / {pair_b} {dir_b} "
        f"z={z:.2f} score={score:.2f} regime={regime}"
    )
    # TODO: place_trade(pair_a, dir_a, lot_size=0.01)
    # TODO: place_trade(pair_b, dir_b, lot_size=0.01)

def ack_signal(signal_id):
    """Mark signal as DISMISSED after acting on it."""
    requests.post(
        f"{MACROFX_URL}/api/hedge-signals/ack",
        json={"id": signal_id, "status": "DISMISSED"},
        timeout=5
    )

while True:
    try:
        data = fetch_signals()
        active = [s for s in data.get("signals", [])
                  if s["status"] == "ACTIVE"]
        for sig in active:
            handle_signal(sig)
            ack_signal(sig["id"])
    except Exception as e:
        logging.error(f"Poll error: {e}")
    time.sleep(POLL_INTERVAL)
"""
MT5 + MacroFX Hedge Bot skeleton.
Extend handle_signal() to place hedged trades via MetaTrader5.
"""
import MetaTrader5 as mt5
import requests, time, logging

MACROFX_URL = "https://your-railway-url.up.railway.app"
LOT_SIZE    = 0.01
MAGIC       = 88001  # unique magic number

MT5_PAIRS = {
    "EURUSD": "EURUSD", "USDJPY": "USDJPY",
    "USDCHF": "USDCHF", "GBPUSD": "GBPUSD",
    # Add all pairs your broker supports
}

def place_trade(symbol, direction, lot):
    mt5_sym = MT5_PAIRS.get(symbol)
    if not mt5_sym: return False
    info = mt5.symbol_info_tick(mt5_sym)
    if not info: return False
    order_type = mt5.ORDER_TYPE_BUY if direction == "LONG" \
                 else mt5.ORDER_TYPE_SELL
    price = info.ask if direction == "LONG" else info.bid
    req = {
        "action": mt5.TRADE_ACTION_DEAL,
        "symbol": mt5_sym, "volume": lot,
        "type": order_type, "price": price,
        "magic": MAGIC, "comment": "hedge_sig",
        "type_time": mt5.ORDER_TIME_GTC,
        "type_filling": mt5.ORDER_FILLING_IOC,
    }
    result = mt5.order_send(req)
    return result.retcode == mt5.TRADE_RETCODE_DONE

def handle_signal(sig):
    ok_a = place_trade(sig["pair_a"], sig["direction_a"], LOT_SIZE)
    ok_b = place_trade(sig["pair_b"], sig["direction_b"], LOT_SIZE)
    logging.info(
        f"Placed {sig['pair_a']} {sig['direction_a']}={ok_a} | "
        f"{sig['pair_b']} {sig['direction_b']}={ok_b}"
    )

if not mt5.initialize(): raise RuntimeError("MT5 init failed")

while True:
    r = requests.get(f"{MACROFX_URL}/api/hedge-signals", timeout=10)
    for sig in r.json().get("signals", []):
        if sig["status"] == "ACTIVE":
            handle_signal(sig)
            requests.post(f"{MACROFX_URL}/api/hedge-signals/ack",
                json={"id": sig["id"], "status": "DISMISSED"})
    time.sleep(30 * 60)
